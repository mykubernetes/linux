#!/bin/bash

oldpasswd="oldpasswd"
newpasswd="newpasswd"

for i in `cat $(pwd)/host_ip.log`
do

/usr/bin/expect << EOF

 spawn  /usr/bin/sshpass -p "${oldpasswd}"  ssh -l deployer $i
 
 expect {
 "yes/no" { send "yes\r" }
 }
 
 expect "current"
 send "$oldpasswd\r"
 
 expect "New"
 send "$newpasswd\r"
 
 expect "Retype"
 send "$newpasswd\r"   
 send "exit\r"
 expect eof
EOF
done
